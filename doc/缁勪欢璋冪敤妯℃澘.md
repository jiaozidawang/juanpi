## 滚动组件 BScroll

##### template
```
<scroll
    class="wrapper"
    :pullup="pullup"
    :data="dataList"
    :pulldown="pulldown"
    @pulldown="refreshData"
    @scrollToEnd="loadMore"
>
    <ul>
        <!-- ----这里写你的子项---- -->
        <div v-for="(item, index) in dataList" :keys="index">{{item}}</div>
        <!-- ----这里写你的子项---- -->

        <div v-if="dataList.length <= count && isLoading" class="btn-box">正在加载中...</div>
        <div v-if="dataList.length >= count" class="btn-box">没有更多了</div>
    </ul>
</scroll>
```

##### script

```
import scroll from '@comps/BScroll';
export default {
  name: "better-demo",
  components: {
    scroll
  },
  data () {
    return {
      pulldown: true, // 打开派发下拉事件
      pullup: true, // 打开派发到底部事件
      count: 15, // 数据总数量
      isLoading: false, // 是否显示正在加载中
      dataList: [] // 数据数组
    }
  },
  methods: {

    // 上拉加载更多方法
    loadMore () {
      // 当数组长度大于数据库数据总条数，关闭上拉监听
      if(this.dataList.length >= this.count) {
        this.pullup = false;
        return;
      }
      this.isLoading = true; // 显示正在加载中提示,需要在数据请求完后设置成false

      // 请求数据的方法,在这个方法里面
      this.loadData()
    },

    // 下拉刷新方法
    refreshData () {
      this.dataList = []; // 清空数据
      /**
       * 模仿真实请求
       */
      this.isLoading = true; // 显示正在加载中,需要在数据请求完后设置成false

      // 请求数据的方法,在这个方法里面
      this.loadData()
    },
    // 请求数据的方法
    loadData () {
        this.$tools.rpc().then(res => {
            this.isLoading = false

            // 把请求回来的数据push进this.dataList数组里面
        )
    }
  }
};
</script>
<style lang="less" scoped>
.btn-box{
  text-align: center;
}
</style>
```
